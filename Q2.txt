module Q2 (address, c, s,d);

 input c,s;

 input [15:0] d;

 output reg[15:0] address;

begin 

 address = 16'h0000;

end

always @(*) begin

 if ( c == 1 && s == 1 ) begin

   address = d  - 1;

 end

 else if (c == 1 && s == 0) begin

   address = d + 1;

 end

 else
 
   address = d;

 end

endmodule
